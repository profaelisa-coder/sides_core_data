function updateStaffFee() { const t = tournamentCatalog[document.getElementById('staffSelectTourney').value]; if(t) document.getElementById('staffFee').value = t.fee; } function resetSystem() { if(confirm("DELETE EVERYTHING?") && prompt("Type DELETE:") === "DELETE") location.reload(); } // A lógica matemática do Payout entrará aqui no próximo passo! // --- MOTOR MATEMÁTICO V3.0 (PADRÃO TPCI) --- function calculatePrizeTickets() { const index = document.getElementById('staffSelectTourney').value; const t = tournamentCatalog[index]; const players = parseInt(document.getElementById('staffPlayers').value) || 0; // Pega as variáveis do Setup do Lead const participationPercent = parseFloat(document.getElementById('masterTax').value) || 20; const factor = parseFloat(document.getElementById('masterFactor').value) || 1.75; if (!t || players < 4) { alert("Please select a tournament and enter at least 4 players!"); return null; } // 1. CÁLCULO DO LIMITE MÁXIMO (Custo Material subtraído antes de tudo) const netPoolCash = (players * t.fee) - (players * t.cost); const maxTicketsAllowed = Math.floor(netPoolCash * 10); // 2. PRÊMIO DE PARTICIPAÇÃO (O valor "Participation %" dividido por todos) const participationPool = (netPoolCash * (participationPercent / 100)) * 10; const participationEach = Math.floor((participationPool / players) / 10) * 10; const totalParticipationDistributed = participationEach * players; // 3. POTE DE RANKING (O que sobra para o Top) const rankingPool = maxTicketsAllowed - totalParticipationDistributed; // 4. TABELA OFICIAL TPCI (Cortes de Rank e Rodadas) let topRank = 2; let rounds = 3; if (players >= 9 && players <= 16) { topRank = 4; rounds = 4; } else if (players >= 17 && players <= 32) { topRank = 8; rounds = 5; } else if (players >= 33 && players <= 64) { topRank = 16; rounds = 6; } else if (players >= 65 && players <= 128) { topRank = 32; rounds = 7; } else if (players >= 129 && players <= 226) { topRank = 64; rounds = 8; } else if (players >= 227) { topRank = 128; rounds = 9; } // 5. DISTRIBUIÇÃO CURVA (Spread Factor) let weights = []; let sumWeights = 0; for (let i = 0; i < topRank; i++) { let w = Math.pow(topRank - i, factor); weights.push(w); sumWeights += w; } // Calcula prêmios brutos e arredonda de 10 em 10 let finalPayouts = weights.map(w => Math.round(((w / sumWeights) * rankingPool) / 10) * 10); // 6. TRAVA DE SEGURANÇA (Diferença tirada do 1º e 2º lugar se estourar o limite) let currentTotal = finalPayouts.reduce((a, b) => a + b, 0) + totalParticipationDistributed; let diff = currentTotal - maxTicketsAllowed; if (diff > 0) { const halfDiff = Math.ceil((diff / 2) / 10) * 10; finalPayouts[0] -= halfDiff; finalPayouts[1] -= (diff - halfDiff); } return { tournament: t, players: players, rounds: rounds, participation: participationEach, topRank: topRank, payouts: finalPayouts }; } function generatePayout() { const res = calculatePrizeTickets(); if (!res) return; // Preenche o Cabeçalho do PDF (SUN4 Style) document.getElementById('outEventName').innerText = document.getElementById('masterEvent').value; document.getElementById('outTourneyName').innerText = res.tournament.name; document.getElementById('outDetails').innerText = `${res.tournament.day.toUpperCase()} - ${res.tournament.hour} | Players: ${res.players} | Rounds: ${res.rounds}`; // Preenche a Tabela de Ranks agrupando (1st, 2nd, 3rd-4th...) const tbody = document.getElementById('payoutBody'); tbody.innerHTML = ""; const labels = ["1st Place", "2nd Place", "3rd - 4th", "5th - 8th", "9th - 16th", "17th - 32nd", "33rd - 64th", "65th - 128th"]; const steps = [1, 1, 2, 4, 8, 16, 32, 64]; // Quantos jogadores em cada grupo let currentIdx = 0; for (let i = 0; i < steps.length; i++) { if (currentIdx >= res.topRank) break; tbody.innerHTML += ` ${labels[i]} ${res.payouts[currentIdx].toLocaleString()} `; currentIdx += steps[i]; } // Linha Final de Participação const partRankLabel = res.topRank === 2 ? "3rd" : (res.topRank === 4 ? "5th" : (res.topRank === 8 ? "9th" : (res.topRank === 16 ? "17th" : "33rd"))); tbody.innerHTML += ` ${partRankLabel}+ (Participation) ${res.participation.toLocaleString()} `; // Mostra a área de impressão document.getElementById('printArea').style.display = 'block'; window.scrollTo(0, document.body.scrollHeight); } // Função vazia para não dar erro no botão enquanto não fazemos o reporte consolidado function showFinalReport() { alert("Generating Final Report...
